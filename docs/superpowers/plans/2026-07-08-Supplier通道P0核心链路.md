```php
// app/Api/Supplier/Middleware/CheckAuth.php
<?php
declare(strict_types=1);

namespace App\Api\Supplier\Middleware;

use App\Enums\RoleEnum;
use App\Modules\Merchant\Models\Merchant;
use Closure;
use Illuminate\Http\Request;
use Symfony\Component\HttpFoundation\Response;

class CheckAuth
{
    public function handle(Request $request, Closure $next): Response
    {
        $staff = auth('merchant_staff')->user();

        if ($staff === null) {
            return response()-&gt;json(['code' =&gt; 401, 'message' =&gt; '未登录', 'data' =&gt; null], 401);
        }

        $merchant = Merchant::find($staff->merchant_id);

        if ($merchant === null || (int) $merchant->status !== 1) {
            return response()-&gt;json(['code' =&gt; 403, 'message' =&gt; '商户状态异常', 'data' =&gt; null], 403);
        }

        // 校验商户类型为供应商
        $type = $merchant->type ?? $merchant->role_type ?? null;
        if ($type !== RoleEnum::Supplier->value) {
            return response()-&gt;json(['code' =&gt; 403, 'message' =&gt; '非供应商账号', 'data' =&gt; null], 403);
        }

        $request->attributes->set('merchant_id', $staff->merchant_id);

        return $next($request);
    }
}
```

```php
// app/Api/Supplier/Controllers/AuthController.php
public function login(LoginRequest $request): JsonResponse
{
    $credentials = $request->only('username', 'password');
    if (! $token = auth('merchant_staff')->attempt($credentials)) {
        return $this->error('登录失败', 2001);
    }

    $staff = auth('merchant_staff')->user();
    $merchant = Merchant::find($staff->merchant_id);

    if ($merchant === null || $merchant->type !== RoleEnum::Supplier->value) {
        auth('merchant_staff')->logout();
        return $this->error('非供应商账号', 2002);
    }

    return $this->success([
        'access_token' =&gt; $token,
        'token_type' =&gt; 'Bearer',
        'expires_in' =&gt; auth('merchant_staff')->factory()->getTTL() * 60,
    ]);
}

public function logout(): JsonResponse
{
    auth('merchant_staff')->logout();
    return $this->success(['message' =&gt; '退出成功']);
}
```
